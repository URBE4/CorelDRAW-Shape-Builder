Private Function FM() As Long
    On Error Resume Next
    Dim ix As Long
    ix = cboFType.ListIndex
    If ix < 0 Then FM = 1: Exit Function
    Dim ub As Long, lb As Long
    ub = UBound(mPresetFM)
    lb = LBound(mPresetFM)
    If Err.Number <> 0 Then
        Err.Clear
        FM = ListIndexToFM(ix)
        Exit Function
    End If
    If ix <= ub And ix >= lb Then
        FM = mPresetFM(ix)
    Else
        FM = ListIndexToFM(ix)
    End If
End Function

Private Sub RebuildFacadeTypeCombo()
    On Error Resume Next
    Dim p As String, fh As Integer, line As String
    Dim parts() As String
    Dim i As Long, n As Long
    Dim csvCnt As Long
    Dim csvLines() As String
    cboFType.Clear
    csvCnt = 0
    p = Module1.FacadeTypesCsvPath()
    If Dir(p) <> "" Then
        fh = FreeFile
        Open p For Input As #fh
        Line Input #fh, line
        Do While Not EOF(fh)
            Line Input #fh, line
            line = Trim$(line)
            If Len(line) > 0 Then
                If InStr(line, ";") > 0 Then
                    parts = Split(line, ";")
                    If UBound(parts) >= 1 Then
                        cboFType.AddItem Trim$(parts(0))
                        ReDim Preserve csvLines(0 To csvCnt)
                        csvLines(csvCnt) = line
                        csvCnt = csvCnt + 1
                    End If
                End If
            End If
        Loop
        Close #fh
    End If
    n = csvCnt
    If n > 0 Then
        ReDim mPresetFM(0 To n - 1)
        ReDim mPresetRows(0 To n - 1)
        For i = 0 To n - 1
            parts = Split(csvLines(i), ";")
            Dim k As Long
            k = CLng(Val(parts(1)))
            If k < 0 Or k > 3 Then k = 0
            mPresetFM(i) = ListIndexToFM(k)
            mPresetRows(i) = csvLines(i)
        Next i
    Else
        ReDim mPresetFM(0 To 0)
        ReDim mPresetRows(0 To 0)
        mPresetFM(0) = 1
        mPresetRows(0) = vbNullString
    End If
    On Error GoTo 0
End Sub

' Тот же формат, что CSV → / ← (H;W;Qty). Возвращает True, если загружена хотя бы одна строка.
Private Function ApplyFacadesFromCsv(ByVal txt As String) As Boolean
    On Error GoTo EH
    Dim lines() As String
    Dim i As Long
    Dim line As String
    Dim sep As String
    Dim parts() As String
    Dim nh As Double, nw As Double
    Dim nq As Long
    Dim cnt As Long
    txt = Replace(txt, vbCrLf, vbLf)
    txt = Replace(txt, vbCr, vbLf)
    lines = Split(txt, vbLf)
    cnt = 0
    For i = 0 To UBound(lines)
        line = Trim$(lines(i))
        If Len(line) = 0 Then GoTo NextI
        If InStr(1, line, ";", vbBinaryCompare) > 0 Then
            sep = ";"
        ElseIf InStr(1, line, ",", vbBinaryCompare) > 0 Then
            sep = ","
        Else
            GoTo NextI
        End If
        If InStr(1, LCase$(line), "h_mm", vbTextCompare) > 0 Then GoTo NextI
        parts = Split(line, sep)
        If UBound(parts) < 2 Then GoTo NextI
        nh = Val(Trim$(parts(0)))
        nw = Val(Trim$(parts(1)))
        nq = CLng(Val(Trim$(parts(2))))
        If nh <= 0 Or nw <= 0 Then GoTo NextI
        If nq < 1 Then nq = 1
        If nq > 500 Then nq = 500
        cnt = cnt + 1
NextI:
    Next i
    If cnt = 0 Then ApplyFacadesFromCsv = False: Exit Function
    mFC = 0
    ReDim mFH(0 To cnt - 1)
    ReDim mFW(0 To cnt - 1)
    ReDim mFQ(0 To cnt - 1)
    cnt = 0
    For i = 0 To UBound(lines)
        line = Trim$(lines(i))
        If Len(line) = 0 Then GoTo NextJ
        If InStr(1, line, ";", vbBinaryCompare) > 0 Then
            sep = ";"
        ElseIf InStr(1, line, ",", vbBinaryCompare) > 0 Then
            sep = ","
        Else
            GoTo NextJ
        End If
        If InStr(1, LCase$(line), "h_mm", vbTextCompare) > 0 Then GoTo NextJ
        parts = Split(line, sep)
        If UBound(parts) < 2 Then GoTo NextJ
        nh = Val(Trim$(parts(0)))
        nw = Val(Trim$(parts(1)))
        nq = CLng(Val(Trim$(parts(2))))
        If nh <= 0 Or nw <= 0 Then GoTo NextJ
        If nq < 1 Then nq = 1
        If nq > 500 Then nq = 500
        mFH(cnt) = nh
        mFW(cnt) = nw
        mFQ(cnt) = nq
        cnt = cnt + 1
NextJ:
    Next i
    mFC = cnt
    RefreshList
    UpdatePreview
    ApplyFacadesFromCsv = True
    Exit Function
EH:
    ApplyFacadesFromCsv = False
End Function

Private Function ParseNeoSegmentsToModule(ByRef errMsg As String) As Boolean
    On Error GoTo EH
    Dim nWant As Long
    Dim raw As String, parts() As String
    Dim i As Long, cnt As Long, t As String
    Dim vals(1 To 8) As Double

    errMsg = vbNullString
    ParseNeoSegmentsToModule = False

    nWant = CLng(Me.Controls("cboSegN").ListIndex) + 2
    If nWant < 2 Or nWant > 8 Then
        errMsg = "Выберите число сегментов от 2 до 8."
        Exit Function
    End If

    raw = CT("txtFacadeSegMm")
    raw = Replace$(Replace$(Replace$(raw, vbCrLf, ";"), vbCr, ";"), vbLf, ";")
    raw = Replace$(raw, ",", ";")
    Do While InStr(raw, ";;") > 0
        raw = Replace$(raw, ";;", ";")
    Loop

    parts = Split(raw, ";")
    cnt = 0
    For i = LBound(parts) To UBound(parts)
        t = Trim$(parts(i))
        If Len(t) > 0 Then
            cnt = cnt + 1
            If cnt > 8 Then
                errMsg = "Слишком много значений (максимум 8)."
                Exit Function
            End If
            vals(cnt) = TN(t, -1#)
            If vals(cnt) <= 0# Then
                errMsg = "Сегмент " & CStr(cnt) & ": укажите положительное число, мм."
                Exit Function
            End If
        End If
    Next i

    If cnt <> nWant Then
        errMsg = "Для выбранного числа сегментов введите ровно " & CStr(nWant) & " положительных значений мм."
        Exit Function
    End If

    Module1.AssignNeoProfileSegments vals, nWant
    ParseNeoSegmentsToModule = True
    Exit Function
EH:

' Первые 4 числа из txtFacadeSegMm (как на чертеже сечения); иначе False — берём 57/7/8/12
Private Function NeoParseFourSegMm(ByRef s1 As Double, ByRef s2 As Double, ByRef s3 As Double, ByRef s4 As Double, ByRef patOut As Double) As Boolean
    On Error GoTo EH
    Dim raw As String, parts() As String
    Dim vals(1 To 8) As Double
    Dim i As Long, cnt As Long, t As String
    NeoParseFourSegMm = False
    patOut = 0#
    raw = Trim$(CT("txtFacadeSegMm"))
    raw = Replace$(Replace$(Replace$(raw, vbCrLf, ";"), vbCr, ";"), vbLf, ";")
    raw = Replace$(raw, ",", ";")
    Do While InStr(raw, ";;") > 0
        raw = Replace$(raw, ";;", ";")
    Loop
    parts = Split(raw, ";")
    cnt = 0
    For i = LBound(parts) To UBound(parts)
        t = Trim$(parts(i))
        If Len(t) > 0 Then
            cnt = cnt + 1
            If cnt > 8 Then Exit For
            vals(cnt) = TN(t, -1#)
            If vals(cnt) <= 0# Then GoTo EH
        End If
    Next i
    If cnt < 4 Then Exit Function
    s1 = vals(1): s2 = vals(2): s3 = vals(3): s4 = vals(4)
    patOut = s1 + s2 + s3 + s4
    NeoParseFourSegMm = (patOut > 0.01)
    Exit Function
EH:
    NeoParseFourSegMm = False
End Function

Private Function NeoFmtMm(ByVal v As Double) As String
    On Error Resume Next
    If Abs(v - CLng(v)) < 0.06 Then
        NeoFmtMm = CStr(CLng(v))
    Else
        NeoFmtMm = Replace(Format$(v, "0.0"), ",", ".")
    End If
End Function

' SVG: сечение штура с узлами как на фрезе (4 сегмента), размеры + выносные линии

    Dim pat As Double
    Dim j1 As Double, j2 As Double, j3 As Double, j4 As Double

    Dim t As String
    Dim xs As String
    If Not NeoParseFourSegMm(s1, s2, s3, s4, pat) Then
        s1 = 57#: s2 = 7#: s3 = 8#: s4 = 12#
        pat = s1 + s2 + s3 + s4
    End If
    j1 = s1
    j2 = s1 + s2
    j3 = s1 + s2 + s3
    j4 = pat
    If s2 > 0.0001 Then
        xMid = s1 + (3# / 7#) * s2
    Else
        xMid = j1
    End If
    d = "M0,4 L" & NeoFmtMm(j1) & ",4 L" & NeoFmtMm(xMid) & ",7 L" & NeoFmtMm(j2) & ",5.2 L" & NeoFmtMm(j3) & ",8 L" & NeoFmtMm(j4) & ",4 L" & NeoFmtMm(j4) & ",12 L0,12 Z"
    xs = Replace(Format$(pat, "0.###"), ",", ".")
    t = "<svg xmlns=""http://www.w3.org/2000/svg"" width=""100%"" height=""100%"" viewBox=""0 0 " & xs & " 40"" preserveAspectRatio=""xMidYMid meet"" style=""background:#E8EBED;border:1px solid #969FA5;border-radius:2px;"">"

    t = t & "<line x1=""" & NeoFmtMm(j2) & """ y1=""12"" x2=""" & NeoFmtMm(j2) & """ y2=""25"" stroke-dasharray=""0.4 0.4""/>"
    t = t & "<line x1=""" & NeoFmtMm(j3) & """ y1=""12"" x2=""" & NeoFmtMm(j3) & """ y2=""25"" stroke-dasharray=""0.4 0.4""/>"
    t = t & "<line x1=""" & NeoFmtMm(j4) & """ y1=""12"" x2=""" & NeoFmtMm(j4) & """ y2=""25"" stroke-dasharray=""0.4 0.4""/>"
    t = t & "</g>"
    t = t & "<g stroke=""#1f2a33"" stroke-width=""0.14"" fill=""#1f2a33"" font-size=""3.1"" font-family=""Segoe UI,sans-serif"">"
    t = t & "<line x1=""0"" y1=""26"" x2=""" & NeoFmtMm(pat) & """ y2=""26""/>"
    t = t & "<line x1=""0"" y1=""24.8"" x2=""0"" y2=""27.2""/><line x1=""" & NeoFmtMm(j1) & """ y1=""24.8"" x2=""" & NeoFmtMm(j1) & """ y2=""27.2""/>"
    t = t & "<line x1=""" & NeoFmtMm(j2) & """ y1=""24.8"" x2=""" & NeoFmtMm(j2) & """ y2=""27.2""/><line x1=""" & NeoFmtMm(j3) & """ y1=""24.8"" x2=""" & NeoFmtMm(j3) & """ y2=""27.2""/>"
    t = t & "<line x1=""" & NeoFmtMm(j4) & """ y1=""24.8"" x2=""" & NeoFmtMm(j4) & """ y2=""27.2""/>"
    t = t & "<text x=""" & NeoFmtMm(s1 / 2#) & """ y=""33.5"" text-anchor=""middle"">" & NeoFmtMm(s1) & "</text>"
    t = t & "<text x=""" & NeoFmtMm(s1 + s2 / 2#) & """ y=""33.5"" text-anchor=""middle"">" & NeoFmtMm(s2) & "</text>"

    t = t & "<text x=""" & NeoFmtMm(s1 + s2 + s3 + s4 / 2#) & """ y=""33.5"" text-anchor=""middle"">" & NeoFmtMm(s4) & "</text>"
    t = t & "</g>"
    t = t & "<text x=""" & NeoFmtMm(pat / 2#) & """ y=""4.5"" text-anchor=""middle"" fill=""#3D4A54"" font-size=""2.5"" font-family=""Segoe UI,sans-serif"">сечение верхнего штура</text>"

    t = t & "</svg>"
    BuildNeoProfileSectionHtml = t

Private Function Build3dHtmlFacade(ByVal hMm As Double, ByVal wMm As Double, ByVal styleIdx As Long) As String
    Dim mx As Double
    If hMm > wMm Then mx = hMm Else mx = wMm
    If mx < 1 Then mx = 1
    Dim avail As Double
    avail = CDbl(PREVIEW_3D_PX) - 14#
    If avail < 18 Then avail = 18
    Dim k As Double
    k = avail / mx
    Dim pxH As Long, pxW As Long
    pxH = CLng(k * hMm)
    pxW = CLng(k * wMm)
    If pxH < 12 Then pxH = 12
    If pxW < 12 Then pxW = 12
    If pxH > CLng(avail) - 1 Then pxH = CLng(avail) - 1
    If pxW > 220 Then pxW = 220
    Dim s As String
    Dim isClassic As Boolean
    Dim isCl2 As Boolean
    Dim isNeo As Boolean
    isNeo = (styleIdx = 3)
    isClassic = (styleIdx < 2) And Not isNeo
    isCl2 = (styleIdx = 1)
    s = "<!DOCTYPE html><html><head><meta http-equiv=""X-UA-Compatible"" content=""IE=edge""/>"
    s = s & "<style>html,body{height:100%;margin:0;overflow:hidden;background:#E5E9EC;color:#323D43;font:8px 'Segoe UI',sans-serif;}"
    s = s & "#box{display:flex;flex-direction:column;height:100%;box-sizing:border-box;}"
    s = s & "#cap{padding:2px 4px 0;text-align:center;line-height:9px;white-space:nowrap;overflow:hidden;text-overflow:ellipsis;color:#5E6B73;}"
    s = s & "#stage{flex:1;min-height:0;display:flex;align-items:center;justify-content:center;perspective:220px;padding:0 2px 2px;}"
    s = s & ".door{position:relative;background:linear-gradient(145deg,#B2B9BD,#5E6B73);"
    s = s & "box-shadow:0 3px 10px rgba(50,61,67,.35);transform:rotateX(10deg) rotateY(-16deg);"
    s = s & "border:1px solid #969FA5;box-sizing:border-box;}"
    s = s & ".door.cls{border-radius:2px;box-shadow:0 3px 10px rgba(50,61,67,.35),inset 0 0 0 1px #CED4D9,inset 0 0 0 3px #323D43;}"
    s = s & ".inset{position:absolute;background:rgba(0,0,0,.12);border:1px solid rgba(255,255,255,.08);}"
    s = s & ".arc{position:absolute;left:12%;right:12%;border:1px solid rgba(200,175,150,.35);border-bottom:none;opacity:.88;}"
    s = s & "#neoSec{flex:1;min-height:0;padding:2px;display:flex;align-items:center;justify-content:center;}"
    s = s & "</style></head><body><div id=""box"">"
    s = s & "<div id=""cap"">" & FacadeTypeCaption() & " · H" & CStr(CLng(hMm)) & "×W" & CStr(CLng(wMm)) & " mm</div>"
    If isNeo Then
        s = s & "<div id=""neoSec"">" & BuildNeoProfileSectionHtml() & "</div></div></body></html>"
        Build3dHtmlFacade = s
        Exit Function
    End If
    s = s & "<div id=""stage""><div class=""door"
    If isClassic Then s = s & " cls"
    s = s & """ style=""width:" & CStr(pxW) & "px;height:" & CStr(pxH) & "px;"">"
    If isClassic Then
        If isCl2 Then
            s = s & "<div class=""arc"" style=""top:10%;height:20%""></div>"
            s = s & "<div class=""inset"" style=""top:32%;left:11%;right:11%;bottom:16%""></div>"
        Else
            s = s & "<div class=""arc"" style=""top:8%;height:18%""></div>"
            s = s & "<div class=""inset"" style=""top:30%;left:10%;right:10%;bottom:14%""></div>"
        End If
    Else
        s = s & "<div class=""inset"" style=""top:12%;left:10%;right:10%;bottom:12%""></div>"
    End If
    s = s & "</div></div></div></body></html>"
    Build3dHtmlFacade = s
End Function

    If Len(nm) = 0 Then Exit Sub
    Dim kindIdx As Long
    If cboFType.ListIndex >= 0 And cboFType.ListIndex <= UBound(mPresetFM) Then
        kindIdx = FMToListIndex(mPresetFM(cboFType.ListIndex))
    Else
        kindIdx = FMToListIndex(FM())
    End If
    Module1.UpsertFacadeTypeRow BuildFacadePresetCsvLine(nm, kindIdx)
    RebuildFacadeTypeCombo
    Dim i As Long
    For i = 0 To cboFType.ListCount - 1
        If StrComp(cboFType.List(i), nm, vbTextCompare) = 0 Then
            cboFType.ListIndex = i
            Exit For
        End If
    Next i
    mPrevFTypeIdx = cboFType.ListIndex
    ApplyPresetFromListIndex cboFType.ListIndex
    UpdateFacadeParamPanels
    UpdatePreview
    RefreshReadyStatus
    SetStatus "Тип сохранён: " & nm
    Exit Sub
EH:
    MsgBox "Сохранить тип: " & Err.Description, vbExclamation
End Sub

Private Sub BuildFacade(ByVal closeAfter As Boolean)
    If mFC = 0 Then
        MsgBox "Добавь фасады", vbExclamation
        Exit Sub
    End If

    Dim ii As Long
    For ii = 0 To mFC - 1
        If mFW(ii) <= 0 Or mFH(ii) <= 0 Or mFQ(ii) <= 0 Then
            MsgBox "Позиция " & CStr(ii + 1) & ": H, W и количество должны быть больше 0.", vbExclamation
            Exit Sub
        End If
    Next ii

    Dim nm As Long
    nm = 0
    On Error Resume Next
    nm = Me.Controls("cboNestMode").ListIndex
    On Error GoTo 0

    Dim shW As Double, shH As Double
    Dim allowRot As Boolean, allowMir As Boolean, grainL As Boolean
    On Error Resume Next
    allowRot = CBool(Me.Controls("chkRotate").Value)
    allowMir = CBool(Me.Controls("chkMirror").Value)
    grainL = CBool(Me.Controls("chkGrain").Value)
    On Error GoTo 0

    Dim st As String
    st = CT("txtSheetStock")

    On Error Resume Next
    Module1.mCncLayers = CBool(Me.Controls("chkCncLayers").Value)
    Module1.mGuillotineTallRightStrip = CBool(Me.Controls("chkGuillotineTallRight").Value)
    On Error GoTo 0

    Module1.mFormFaceKind = FM()

    ApplyContourUiToLegacy
    Module1.SetDrawStyleFromForm CV("cboBorder"), CV("cboOrn")
    If FM() = 3 Then
        Dim neoErr As String
        If Not ParseNeoSegmentsToModule(neoErr) Then
            MsgBox neoErr, vbExclamation, FacadeTypeCaption()
            Exit Sub
        End If
        Module1.SetPanelGeometry 0#, 0#, 0#, 0#, 0, 0#, 0, 0#
    Else
        Module1.SetPanelGeometry TN(CT("txtInsetX"), 70), TN(CT("txtInsetY"), 70), _
            TN(CT("txtArcTop"), 0), TN(CT("txtArcSide"), 0), _
            OffsetQtyLong("cboOuterQty", 0), TN(CT("txtOuter"), 23), _
            OffsetQtyLong("cboInnerQty", 1), TN(CT("txtInner"), 20)
    End If

    Dim gapDef As Double
    Dim gapUse As Double
    If nm = 0 Or nm = 2 Then
        shW = TN(CT("txtSheetW"), 2070)
        shH = TN(CT("txtSheetH"), 2800)
        gapDef = 4
    Else
        shW = TN(CT("txtSheetW"), 2070)
        shH = TN(CT("txtSheetH"), 2800)
        gapDef = 25
    End If
    gapUse = TN(CT("txtGap"), gapDef)
    If shW <= 0 Or shH <= 0 Then
        MsgBox "Укажите положительные размеры листа (В и Ш), мм.", vbExclamation
        Exit Sub
    End If
    If gapUse < 0 Then
        MsgBox "Отступ (зазор) не может быть отрицательным.", vbExclamation
        Exit Sub
    End If
    Module1.RunPlacement nm, shW, shH, gapUse, mFC, mFW, mFH, mFQ, allowRot, allowMir, grainL, st

    Dim totalQ As Long, i As Long
    Dim totalAreaM2 As Double
    Dim ps As Double, pm2 As Double, cst As Double
    totalQ = 0
    totalAreaM2 = 0
    For i = 0 To mFC - 1
        totalQ = totalQ + mFQ(i)
        totalAreaM2 = totalAreaM2 + (mFW(i) * mFH(i) * mFQ(i)) / 1000000#
    Next i

    ps = TN(CT("txtPriceSheet"), 0)
    pm2 = TN(CT("txtPriceM2"), 0)
    cst = mSheets * ps + totalAreaM2 * pm2
    mCostTotal = cst

    Dim stat As String
    Dim shWa As Double, shHa As Double
    Dim sheetMm2 As Double, utilPct As Double
    stat = "Листов: " & mSheets & _
           " | Размещено: " & mPlaced & "/" & totalQ & _
           " | Не влезло: " & mSkipped
    shWa = TN(CT("txtSheetW"), 2070)
    shHa = TN(CT("txtSheetH"), 2800)
    sheetMm2 = shWa * shHa * mSheets
    If sheetMm2 > 0 And mSheets > 0 Then
        utilPct = 100# * mUsedArea / sheetMm2
        If utilPct > 100# Then utilPct = 100#
        stat = stat & " | Утилизация " & Format$(utilPct, "0") & "%"
    End If
    If Module1.mCncLayers Then stat = stat & " | ЧПУ: CUT/TEXT"
    Me.Controls("lblStat").Caption = stat
    Me.Controls("lblCostResult").Caption = "Стоимость: " & Format$(cst, "0.00") & " руб  |  S деталей " & Format$(totalAreaM2, "0.00") & " м²"

    If closeAfter Then Unload Me
End Sub
